sectionMany :: String -> CharParser () a -> CharParser () [a] Source #

Parse many elements in a section: <SectionName \n {many parser} \n />.

Examples

Expand
 sectionMany "TokenName" (spaces *> string "bb" <* spaces) will parse:
 <TokenName
     bb bb bb 
     bb
 />
and returns ["bb", "bb", "bb", "bb"]

sectionManyWithKey :: String -> CharParser () a -> CharParser () key -> CharParser () (key, [a]) Source #

Parse many elements in a section with key: <SectionName n {key parser} {many parser} />.

Examples

Expand
 sectionManyWithKey "TokenName" (spaces *> string "bb" <* spaces) (string "key=1n") will parse:
 <TokenName
     key=1
     bb bb bb 
     bb
 />
and returns ("key=1n", ["bb", "bb", "bb", "bb"])

readOption :: String -> CharParser () String Source #
